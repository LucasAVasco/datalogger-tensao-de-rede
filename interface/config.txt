# Modelo para configurar gráficos
{
	title: Título           # Título do gráfico
	day: <ano>/<mês>/<dia>  # Dia dos dados do gráfico
	init: <hora>-<minuto>   # Momento de início dos dados do gráfico
	end: <hora>-<minuto>    # Momento de término dos dados do gráfico
	plotMed: True           # Mostra o valor médio
	plotVpk: True           # Mostra o valor de pico
	plotRms: True           # Mostra o valor RMS
}


# Exemplo de configuração de um gráfico
{
	title: Gráfico de teste
	day: 2021/05/02
	init: 20-38
	end: 20-40
	plotMed: True
	plotVpk: True
}
